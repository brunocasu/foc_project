A. build programs from C file

gcc -Wall -o client  client_a.c -L/usr/lib -lssl -lcrypto -lpthread
gcc -Wall -o server server.c -L/usr/lib -lssl -lcrypto


1. generate private and public keys with password  for users

openssl genrsa -aes256 -passout pass:alicepwd -out privkey.pem 2048
openssl rsa -in privkey.pem -passin pass:alicepwd -pubout -out alice_key.pub


2. generate certificate and pub key for the server

openssl genrsa -out MessageApp_key.pem 3072

openssl req -x509 -new -nodes -key MessageApp_key.pem -sha256 -days 365 -out MessageApp_cert.pem
